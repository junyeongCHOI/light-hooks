# useDebounce

`debounced` 된 `함수`를 반환합니다.

useDebounce(func, wait, options)

wait = number (ms)

options = { leading? : boolean, trailing? : boolean, maxWait? : number }

returns debounced function

# Example

<UseDebounceExample />

# Code

```jsx
import React, { useState } from "react";
import { useDebounce } from "light-hooks";
import styles from "../common.module.scss";

const UseDebounceExample = () => {
  const [value, setValue] = useState("");
  const [debouncedValue, setDebouncedValue] = useState("");
  const debounced = useDebounce((value) => setDebouncedValue(value), 500);

  return (
    <>
      <input
        className={styles.input}
        value={value}
        onChange={(e) => {
          setValue(e.target.value);
          debounced(e.target.value);
        }}
        placeholder="type"
      />
      result: {debouncedValue}
    </>
  );
};

export default UseDebounceExample;
```
